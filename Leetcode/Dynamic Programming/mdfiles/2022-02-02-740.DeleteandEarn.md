class Solution {
public:
    
    int deleteAndEarn(vector<int>& nums) {
        int n = 10001;
       vector<int>bucket(n,0);
        for(int i: nums)
            bucket[i]+=i;
      
        int take=0;
        int skip=0;
        for(int i=0; i<n; i++){
            int curr_skip = max(take,skip);
            int curr_take = bucket[i]+skip;
            take=curr_take;
            skip=curr_skip;
        }
        return max(take,skip);
    }
};
